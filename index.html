/** * ================================================================ * 텀블벅 트래커 - 시트 초기화 스크립트 * ================================================================ * * 사용법: * 1. Apps Script 편집기에서 이 파일 내용을 Code.gs 맨 아래에 붙여넣기 * (또는 별도 파일로 추가해도 OK) * 2. 저장 후 상단 함수 드롭다운에서 실행하고 싶은 함수 선택 * 3. ▶ 실행 버튼 클릭 * 4. 권한 요청이 뜨면 허용 * * 포함된 함수: * - resetAllSheets() : 모든 시트 삭제 후 새로 생성 (⚠️ 데이터 날아감!) * - forceCreateHeaders() : 기존 데이터 유지하고 헤더만 강제로 재생성 * - forceCreateCategorySheet(): categories 시트 강제 생성 * - checkSheetStatus() : 현재 시트 상태 점검 (읽기 전용) */ /** * ⚠️ 모든 시트를 삭제하고 완전히 처음부터 다시 만들기 * 기존 데이터는 모두 사라집니다! */ function resetAllSheets() { const ui = SpreadsheetApp.getUi(); const confirm = ui.alert( '⚠️ 전체 초기화', 'projects 시트와 categories 시트를 모두 삭제하고 새로 만들어요.\n\n기존 데이터는 전부 사라집니다. 계속할까요?', ui.ButtonSet.YES_NO ); if (confirm !== ui.Button.YES) return; const ss = SpreadsheetApp.getActiveSpreadsheet(); // 1. 기존 시트 삭제 const projectsSheet = ss.getSheetByName('projects'); if (projectsSheet) ss.deleteSheet(projectsSheet); const categoriesSheet = ss.getSheetByName('categories'); if (categoriesSheet) ss.deleteSheet(categoriesSheet); // 2. 시트가 최소 1개는 있어야 하므로 임시 시트 확인 if (ss.getSheets().length === 0) { ss.insertSheet('임시'); } // 3. projects 시트 생성 const COLUMNS = [ 'id', 'title', 'creator', 'amount', 'category', 'status', 'paymentDate', 'deliveryDate', 'downloaded', 'downloadedAt', 'sharedWith', 'reward', 'url', 'notes', 'createdAt', 'updatedAt' ]; const projects = ss.insertSheet('projects'); projects.getRange(1, 1, 1, COLUMNS.length).setValues([COLUMNS]); projects.getRange(1, 1, 1, COLUMNS.length) .setFontWeight('bold') .setBackground('#1f1a14') .setFontColor('#fbf7ee'); projects.setFrozenRows(1); projects.setColumnWidth(1, 260); // id projects.setColumnWidth(2, 260); // title projects.setColumnWidth(3, 140); // creator projects.setColumnWidth(4, 100); // amount projects.setColumnWidth(5, 100); // category projects.setColumnWidth(6, 100); // status projects.setColumnWidth(7, 110); // paymentDate projects.setColumnWidth(8, 110); // deliveryDate projects.setColumnWidth(9, 90); // downloaded projects.setColumnWidth(10, 110); // downloadedAt projects.setColumnWidth(11, 160); // sharedWith projects.setColumnWidth(12, 220); // reward projects.setColumnWidth(13, 220); // url projects.setColumnWidth(14, 240); // notes projects.setColumnWidth(15, 180); // createdAt projects.setColumnWidth(16, 180); // updatedAt // 4. categories 시트 생성 const categories = ss.insertSheet('categories'); categories.getRange(1, 1).setValue('name') .setFontWeight('bold') .setBackground('#1f1a14') .setFontColor('#fbf7ee'); categories.setFrozenRows(1); categories.setColumnWidth(1, 240); // 5. 임시 시트 있으면 삭제 const tempSheet = ss.getSheetByName('임시'); if (tempSheet && ss.getSheets().length > 1) { ss.deleteSheet(tempSheet); } ui.alert('✅ 초기화 완료', 'projects 시트와 categories 시트가 새로 만들어졌어요.\n이제 앱에서 새로 데이터를 넣을 수 있습니다.', ui.ButtonSet.OK); } /** * 기존 데이터를 유지한 채 헤더만 강제로 다시 쓰기 * 헤더가 깨졌거나 컬럼 순서가 엉켰을 때 사용 * ※ 데이터 행(2행부터)은 건드리지 않음 */ function forceCreateHeaders() { const COLUMNS = [ 'id', 'title', 'creator', 'amount', 'category', 'status', 'paymentDate', 'deliveryDate', 'downloaded', 'downloadedAt', 'sharedWith', 'reward', 'url', 'notes', 'createdAt', 'updatedAt' ]; const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName('projects'); if (!sheet) { sheet = ss.insertSheet('projects'); } // 현재 시트에 컬럼 수가 부족하면 늘려주기 const currentCols = sheet.getMaxColumns(); if (currentCols < COLUMNS.length) { sheet.insertColumnsAfter(currentCols, COLUMNS.length - currentCols); } // 1행에 헤더 강제 기록 sheet.getRange(1, 1, 1, COLUMNS.length).setValues([COLUMNS]); sheet.getRange(1, 1, 1, COLUMNS.length) .setFontWeight('bold') .setBackground('#1f1a14') .setFontColor('#fbf7ee'); sheet.setFrozenRows(1); // 컬럼 너비 sheet.setColumnWidth(1, 260); sheet.setColumnWidth(2, 260); sheet.setColumnWidth(3, 140); sheet.setColumnWidth(4, 100); sheet.setColumnWidth(5, 100); sheet.setColumnWidth(6, 100); sheet.setColumnWidth(7, 110); sheet.setColumnWidth(8, 110); sheet.setColumnWidth(9, 90); sheet.setColumnWidth(10, 110); sheet.setColumnWidth(11, 160); sheet.setColumnWidth(12, 220); sheet.setColumnWidth(13, 220); sheet.setColumnWidth(14, 240); sheet.setColumnWidth(15, 180); sheet.setColumnWidth(16, 180); try { SpreadsheetApp.getUi().alert( '✅ 헤더 생성 완료', 'projects 시트 1행에 헤더 16개가 기록되었어요.\n기존 데이터 행은 그대로 유지됩니다.', SpreadsheetApp.getUi().ButtonSet.OK ); } catch (e) {} } /** * categories 시트만 강제로 생성 */ function forceCreateCategorySheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName('categories'); if (!sheet) { sheet = ss.insertSheet('categories'); } sheet.getRange(1, 1).setValue('name') .setFontWeight('bold') .setBackground('#1f1a14') .setFontColor('#fbf7ee'); sheet.setFrozenRows(1); sheet.setColumnWidth(1, 240); try { SpreadsheetApp.getUi().alert('✅ categories 시트 준비 완료'); } catch (e) {} } /** * 현재 시트 상태 점검 (읽기 전용, 아무것도 변경하지 않음) * 로그에 현재 헤더와 행 개수를 출력 */ function checkSheetStatus() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const messages = []; // projects 시트 점검 const projects = ss.getSheetByName('projects'); if (!projects) { messages.push('❌ projects 시트가 없음'); } else { const lastRow = projects.getLastRow(); const lastCol = projects.getLastColumn(); const headers = projects.getRange(1, 1, 1, lastCol).getValues()[0]; messages.push('✓ projects 시트:'); messages.push(' - 행 개수: ' + (lastRow - 1) + '건'); messages.push(' - 컬럼 수: ' + lastCol); messages.push(' - 헤더:'); headers.forEach((h, i) => { const col = String.fromCharCode(65 + i); // A, B, C... messages.push(` ${col}: ${h || '(비어있음)'}`); }); } messages.push(''); // categories 시트 점검 const categories = ss.getSheetByName('categories'); if (!categories) { messages.push('❌ categories 시트가 없음'); } else { const lastRow = categories.getLastRow(); messages.push('✓ categories 시트:'); messages.push(' - 커스텀 카테고리 개수: ' + Math.max(0, lastRow - 1)); if (lastRow > 1) { const cats = categories.getRange(2, 1, lastRow - 1, 1).getValues().map(r => r[0]).filter(Boolean); messages.push(' - 목록: ' + cats.join(', ')); } } const report = messages.join('\n'); Logger.log(report); try { SpreadsheetApp.getUi().alert('📊 시트 상태', report, SpreadsheetApp.getUi().ButtonSet.OK); } catch (e) { // UI 컨텍스트가 없을 때는 로그로만 } return report; }
